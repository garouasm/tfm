if (NOT TFM_PARTITION_TA_ATTESTATION)
    return()
endif()

cmake_minimum_required(VERSION 3.15)
cmake_policy(SET CMP0079 NEW)

# The name of the target is required to be of the pattern
# tfm_app_rot_partition_x or tfm_psa_rot_partition_x, as it affects how the
# linker script will lay the partition in memory.
add_library(tfm_app_rot_partition_ta_attestation STATIC)

# Add the source files generated by parse tools when building. The intermedia
# file defines the partition stack. The load info file includes the static data
# of the partition.
target_sources(tfm_app_rot_partition_ta_attestation
    PRIVATE
        tfm_ta_attestation.c
        
)

target_sources(tfm_app_rot_partition_ta_attestation
    PRIVATE
        ${CMAKE_BINARY_DIR}/generated/secure_fw/partitions/ta_attestation/auto_generated/intermedia_tfm_ta_attestation.c
)

target_sources(tfm_partitions
    INTERFACE
         ${CMAKE_BINARY_DIR}/generated/secure_fw/partitions/ta_attestation/auto_generated/load_info_tfm_ta_attestation.c
)

 target_sources(tfm_app_rot_partition_ta_attestation 
     INTERFACE 
        # ${CMAKE_BINARY_DIR}/lib/ext/tfm_test_repo-src/test/secure_fw/suites/attestation/attest_token_decode_common.c
        # ${CMAKE_BINARY_DIR}/lib/ext/tfm_test_repo-src/test/secure_fw/suites/attestation/attest_token_decode_asymmetric.c
        # ${CMAKE_BINARY_DIR}/lib/ext/tfm_test_repo-src/test/secure_fw/suites/attestation/attest_token_test.c
        # ${CMAKE_BINARY_DIR}/lib/ext/tfm_test_repo-src/test/secure_fw/suites/attestation/ext/qcbor_util/qcbor_util.c
        ${CMAKE_CURRENT_SOURCE_DIR}/attestation/attest_token_decode_common.c
        ${CMAKE_CURRENT_SOURCE_DIR}/attestation/attest_token_decode_asymmetric.c
        ${CMAKE_CURRENT_SOURCE_DIR}/attestation/attest_token_test.c
        ${CMAKE_CURRENT_SOURCE_DIR}/attestation/ext/qcbor_util/qcbor_util.c

 )
target_include_directories(tfm_app_rot_partition_ta_attestation
    PRIVATE
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}>
    PUBLIC
        ${CMAKE_BINARY_DIR}/generated/secure_fw/partitions/ta_attestation
)

target_include_directories(tfm_partitions
    INTERFACE
        ${CMAKE_BINARY_DIR}/generated/secure_fw/partitions/ta_attestation
)

target_include_directories(tfm_app_rot_partition_ta_attestation
    PUBLIC 
        # ${CMAKE_BINARY_DIR}/lib/ext/tfm_test_repo-src/test/secure_fw/suites/attestation
        # ${CMAKE_BINARY_DIR}/lib/ext/tfm_test_repo-src/test/secure_fw/suites/attestation/secure
        # ${CMAKE_BINARY_DIR}/lib/ext/tfm_test_repo-src/test/secure_fw/suites/attestation/ext/qcbor_util
        ${CMAKE_CURRENT_SOURCE_DIR}/attestation
        ${CMAKE_CURRENT_SOURCE_DIR}/attestation/secure
        ${CMAKE_CURRENT_SOURCE_DIR}/attestation/ext/qcbor_util
    PRIVATE
        .
)


target_link_libraries(tfm_app_rot_partition_ta_attestation
    PRIVATE
        psa_interface
        platform_s
        tfm_sprt
)

target_link_libraries(tfm_app_rot_partition_ta_attestation
    PUBLIC
        tfm_qcbor_s
        tfm_t_cose_s
        tfm_attestation_defs
        tfm_boot_status
        tfm_psa_rot_partition_crypto
        tfm_test_framework_common
)
