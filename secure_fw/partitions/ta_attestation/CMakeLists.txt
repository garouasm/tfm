if (NOT TFM_PARTITION_TA_ATTESTATION)
    return()
endif()

cmake_minimum_required(VERSION 3.15)
cmake_policy(SET CMP0079 NEW)

# The name of the target is required to be of the pattern
# tfm_app_rot_partition_x or tfm_psa_rot_partition_x, as it affects how the
# linker script will lay the partition in memory.
add_library(tfm_app_rot_partition_ta_attestation STATIC)

# Add the source files generated by parse tools when building. The intermedia
# file defines the partition stack. The load info file includes the static data
# of the partition.
target_sources(tfm_app_rot_partition_ta_attestation
    PRIVATE
        tfm_ta_attestation.c
)

target_sources(tfm_app_rot_partition_ta_attestation
    PRIVATE
        ${CMAKE_BINARY_DIR}/generated/secure_fw/partitions/ta_attestation/auto_generated/intermedia_tfm_ta_attestation.c
)

target_sources(tfm_partitions
    INTERFACE
         ${CMAKE_BINARY_DIR}/generated/secure_fw/partitions/ta_attestation/auto_generated/load_info_tfm_ta_attestation.c
)

target_include_directories(tfm_app_rot_partition_ta_attestation
    PRIVATE
        $<BUILD_INTERFACE:${CMAKE_CURRENT_SOURCE_DIR}>
    PUBLIC
        ${CMAKE_BINARY_DIR}/generated/secure_fw/partitions/ta_attestation
)

target_include_directories(tfm_partitions
    INTERFACE
        ${CMAKE_BINARY_DIR}/generated/secure_fw/partitions/ta_attestation
)

target_link_libraries(tfm_app_rot_partition_ta_attestation
    PRIVATE
        psa_interface
        platform_s
        tfm_sprt
)
